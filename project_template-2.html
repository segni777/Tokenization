btc_data = Table.read_table("BTC-USD (20241216000000000 _ 20140930000000000).csv")

# Cleaning and converting 'Close' prices to floats by removing commas
close_prices_str = btc_data.column('Close')  # Extract as strings initially
close_prices_cleaned = [float(price.replace(',', '')) for price in close_prices_str]  # Remove commas and convert to float
btc_data = btc_data.with_column('Close', close_prices_cleaned)


close_prices = btc_data.column('Close')

# Calculate the mean closing price of the last 30 days 
recent_period = close_prices[-30:] 
# We use the mean of the last 30 days as the baseline
hypothesized_mean = np.mean(recent_period)  

# Sample mean and standard deviation for the full dataset
sample_mean = np.mean(close_prices)
sample_std = np.std(close_prices)

n = len(close_prices)
t_statistic = (sample_mean - hypothesized_mean) / (sample_std / np.sqrt(n))
df = n - 1

# Calculating the two-tailed p-value using the cumulative distribution function of the t-distribution
p_value = 2 * (1 - stats.t.cdf(abs(t_statistic), df))

alpha = 0.05
if p_value < alpha:
    conclusion = "Reject the null hypothesis: The mean closing price of Bitcoin has significantly increased."
else:
    conclusion = "Fail to reject the null hypothesis: No significant increase in the mean closing price of Bitcoin."
print(f"T-statistic: {t_statistic}")
print(f"P-value: {p_value}")
print(conclusion)

T-statistic: -11.811490558390169
P-value: 0.0
Reject the null hypothesis: The mean closing price of Bitcoin has significantly increased.

For this section, I chose a one-sample t-test to evaluate whether the mean closing price of bitcoinovet the full dataset is significantly different from the mean clsoing price of the last days. The reason for using a t-test is because the population standard deviation is unknown to us and the dample size is large enough to justify the assumptions of the t-distribution. It's also appropriate because it allows for comparison between the hypothesized mean(last 30 days) and the observed sample mean(full dataset).

The large negative value of the t-statistic suggests a significant downward deviation from the hypothesized mean, meaning the Bitcoin price has increased significantly over time. A t-statistic of this magnitude shows that the observed difference is far beyond what we would expect by random chance.A p-value of 0.0 indicates that the observed difference is virtually impossible under the null hypothesis, providing strong evidence that Bitcoin's mean closing price has significantly increased. btc_data = Table.read_table("BTC-USD (20241216000000000 _ 20140930000000000).csv")

# Cleaning and converting to floats
close_prices_str = btc_data.column('Close')
close_prices_cleaned = [float(price.replace(',', '')) for price in close_prices_str]
btc_data = btc_data.with_column('Close', close_prices_cleaned)
volume_str = btc_data.column('Volume')
volume_cleaned = [float(vol.replace(',', '')) for vol in volume_str]
btc_data = btc_data.with_column('Volume', volume_cleaned)

price_change = np.diff(btc_data.column('Close'))
price_change = np.append(price_change, 0)  # Append 0 to match the length of the original table
btc_data = btc_data.with_column('Price Change', price_change)
btc_data = btc_data.with_column('Upward Movement', (btc_data.column('Price Change') > 0).astype(int))
features = btc_data.drop('Upward Movement').select('Close', 'Volume').to_array()
labels = btc_data.column('Upward Movement')

# Training and testing sets
split_index = int(len(features) * 0.8)
train_features = features[:split_index]
test_features = features[split_index:]
train_labels = labels[:split_index]
test_labels = labels[split_index:]


train_features = np.array([list(row) for row in train_features], dtype=np.float64)
test_features = np.array([list(row) for row in test_features], dtype=np.float64)

if train_features.ndim == 1:
    train_features = train_features.reshape(-1, 1)
if test_features.ndim == 1:
    test_features = test_features.reshape(-1, 1)


knn5 = KNeighborsClassifier(n_neighbors=5)
knn9 = KNeighborsClassifier(n_neighbors=9)
knn15 = KNeighborsClassifier(n_neighbors=15)
knn5.fit(train_features, train_labels)
knn9.fit(train_features, train_labels)
knn15.fit(train_features, train_labels)

# Evaluating the models
threshold_values = np.array([0.1, 0.3, 0.5, 0.7, 0.9])
knn5_recall, knn9_recall, knn15_recall = [], [], []
knn5_precision, knn9_precision, knn15_precision = [], [], []

for threshold in threshold_values:
    # Predictions for each threshold
    pred5 = knn5.predict_proba(test_features)[:, 1] > threshold
    pred9 = knn9.predict_proba(test_features)[:, 1] > threshold
    pred15 = knn15.predict_proba(test_features)[:, 1] > threshold

    # Calculating recall and precision with zero_division=0 to handle cases with no predicted samples
    knn5_recall.append(recall_score(test_labels, pred5, zero_division=0))
    knn9_recall.append(recall_score(test_labels, pred9, zero_division=0))
    knn15_recall.append(recall_score(test_labels, pred15, zero_division=0))

    knn5_precision.append(precision_score(test_labels, pred5, zero_division=0))
    knn9_precision.append(precision_score(test_labels, pred9, zero_division=0))
    knn15_precision.append(precision_score(test_labels, pred15, zero_division=0))

# Plotting precision-recall tradeoff
plt.figure(figsize=(12, 6))

plt.plot(threshold_values, knn9_precision, label="K=9 Precision", linestyle='--', marker='s')
plt.plot(threshold_values, knn15_precision, label="K=15 Precision", linestyle='--', marker='^')
